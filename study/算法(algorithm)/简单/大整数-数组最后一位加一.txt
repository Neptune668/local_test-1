# 给定一个表示 大整数 的整数数组 digits，其中 digits[i] 是整数的第 i 位数字。
# 这些数字按从左到右，从最高位到最低位排列。这个大整数不包含任何前导 0。
# 将大整数加 1，并返回结果的数字数组
class Solution:
    def plusOne(self, digits: list[int]) -> list[int]:
        for i in range(len(digits) - 1, -1, -1):
            if digits[i] < 9:
                digits[i] += 1
                return digits
            digits[i] = 0

        return [1] + digits

sol = Solution()
print(sol.plusOne([1, 2, 9, 9]))
